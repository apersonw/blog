原子更新和聚合查询

发表于 2019-02-12

1.更新对象，为保证原子性，均使用Update
/**

  • 更新商品最低价和最高价
    */
    @Override
    public Mono updateMinAndMaxSalePrice(String tenantId, String id, long minSalePrice, long maxSalePrice) {
    Query query = Query.query(where(“id”).is(id).and(“tenantId”).is(tenantId));
    return reactiveMongoTemplate
    .updateFirst(
        query,
        Update.update("minSalePrice", minSalePrice).set("maxSalePrice", maxSalePrice),
        Goods.class
).map(UpdateResult::wasAcknowledged);
    }

2.聚合查询：
/**

  • 查询指定商品评分和评分人数
    */
    @Override
    public Mono countRating(String goodsId, String tenantId) {
    TypedAggregation ratingTypedAggregation = Aggregation.newAggregation(
    Rating.class,//查询实体
Aggregation.match(Criteria.where("id").ne(null).and("goodsId").is(goodsId).and("tenantId").is(tenantId)),//查询条件
Aggregation.project("goodsScroe"),//查询后需要包含的字段
Aggregation.group()//创建一个新的实例对象使用project中给定的字段
        .count()//统计match查询出来的实体对象数
        .as("shopScore")//将count出来的数存入实体指定字段，此处将评分人数存入
        .sum("goodsScroe")//表达式计算指定字段值得和
        .as("goodsScroe")//将和存入指定字段
    );
    return reactiveMongoTemplate
    .aggregate(ratingTypedAggregation, Rating.class)
.single()//只取第一个查询到对象
.onErrorResume(t -> t instanceof NoSuchElementException, r -> Mono.just(new Rating()))
.map(r -> {
    GoodsRatingModel goodsRatingModel = new GoodsRatingModel();
    goodsRatingModel.setRatingTotalNum(r.getGoodsScroe());
    goodsRatingModel.setRatingUserNum(r.getShopScore());
    return goodsRatingModel;
});
    }
